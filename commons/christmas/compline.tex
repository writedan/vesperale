\vesection{The Nativity of the Lord until the Solemnity of Mary at Compline}

\gresetinitiallines{1}
\gregorioscore{commons/christmas/compline/nativity_antiphon.gabc}

\gresetinitiallines{0}
\veheading{Salvator mundi Domine.}
\gregorioscore{commons/advent/2/compline/hymn.gabc}

\begin{multicols}{2}
	\interlinepenalty=10000
	
	2. Be with us now, in mercy night,\\
	And spare thy servants when they cry,\\
	Our sins blot out, our prayers receive,\\
	Our darkness lighten, and forgive.
	
	3. O let not sleep o'ercome the soul,\\
	Nor Satan with his spirits fouls:\\
	Our flesh keep chaste, that it may be,\\
	A holy temple unto thee.
	
	4. To thee, who dost our souls renew,\\
	With heartfelt vows we humbly sue,\\
	That pure in heart, and free stain,\\
	We from our beds may rise again.
	
	5. All honor, laud, and glory be,\\
	O Jesu, Virgin-born, to thee;\\
	All glory, as is ever meet,\\
	To Father and to Paraclete. Amen.
\end{multicols}

\gresetinitiallines{1}
