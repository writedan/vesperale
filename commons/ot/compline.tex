\veheading{Simple Tone.}

\gregorioscore{commons/ot/compline/hymn.gabc}

\begin{multicols}{2}
	\interlinepenalty=10000
	
	2. From all ill dreams defend our eyes,\\
	From nightly fears and fantasies:\\
	Tread under foot our ghostly foe.\\
	That no pollution we may know.
	
	3. O Father, that we ask be done,\\
	Through Jesus Christ, thine only Son:\\
	Who, with the Holy Ghost and thee,\\
	Doth live and reign eternally. Amen.
	
	\instruct{But on feasts of the B.V.M.}\\
	3. All honor, laud, and glory be,\\
	O Jesu, Virgin-born, to thee:\\
	All glory, as is ever meet,\\
	To Father and to Paraclete. Amen.
	
	\columnbreak
	
	\instruct{Conclude always:}
	
	\gabcsnippet{A(ef)men.(ede)}
\end{multicols}
