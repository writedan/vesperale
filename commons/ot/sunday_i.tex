\gresetinitiallines{0}
\veheading{\textit{Lucis Creator optime}.}
\gregorioscore{sunday/ii/lucis_creator_optime.gabc}

\begin{multicols}{2}
	\interlinepenalty=10000
	\raggedcolumns
	
	2. Whose wisdom joined in meet array\\
	The morn and eve, and named them day:\\
	Night comes with all its darkling fears,\\
	Regard thy people's prayers and tears.
	
	3. Lest, sunk in sin, and whelmed with strife,\\
	They lose the gift of endless life;\\
	While thinking but the thoughts of time,\\
	They weave new chains of woe and crime.
	
	4. But grant them grace that they may strain\\
	The heav'nly gate and prize to gain:\\
	Each harmful lure aside to cast,\\
	And purge away each error past.
	
	\instruct{Until the Purification}\\
	5. All honor, laud, and glory be,\\
	O Jesu, Virgin-born, to thee:\\
	All glory, as is ever meet,\\
	To Father and to Paraclete. Amen.
	
	\instruct{After the Purification}\\
	5. O Father, that we ask be done,\\
	Through Jesus Christ, thine only Son:\\
	Who, with the Holy Ghost and thee,\\
	Doth live and reign eternally. Amen.
\end{multicols}
